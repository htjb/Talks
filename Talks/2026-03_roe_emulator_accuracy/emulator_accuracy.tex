% ── 2. Why emulators? ─────────────────────────────────────────────────────────
\begin{frame}{Emulators in astrophysics and cosmology}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{itemize}
                \item Detailed models take \textbf{hours to evaluate} per parameter set
                      (e.g.\ 21-cm semi-numerical codes, CMB solvers, SED models)
                \item Bayesian inference requires $10^4$--$10^6$ evaluations
                \item Neural network \textbf{emulators} learn the mapping
                      $\theta \mapsto M(\theta)$ from training data and evaluate in
                      \textbf{milliseconds}
                \item Used across the field: 21-cm, CMB, SED fitting, LSS, GWs \ldots
                \item \textbf{Their role will only grow as SKA delivers data at
                      unprecedented scale}
            \end{itemize}
        \end{column}
        \hfill
        \begin{column}{0.42\textwidth}
            \vspace{1em}
            \centering
            % schematic figure -- replace with a nice emulator diagram if available
            \includegraphics[width=\textwidth]{figs/posterior_consistency.png}
            \\[0.3em]
            {\small globalemu vs ARES posteriors\\(full details later)}
        \end{column}
    \end{columns}
\end{frame}

% ── 3. The problem ────────────────────────────────────────────────────────────
\begin{frame}{How accurate does an emulator need to be?}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{itemize}
                \item We measure accuracy in \textbf{signal space}:
                      \[
                          \mathrm{RMSE} = \sqrt{\frac{1}{N_d}\sum_i
                          \bigl(S_\mathrm{true} - S_\mathrm{pred}\bigr)^2}
                      \]
                \item Rules of thumb exist, e.g.\
                      ``$\mathrm{RMSE} \lesssim 0.1\sigma$''
                      (\textsc{globalemu}, Bevins et al.\ 2021)
                \item But \textbf{what does this mean for the recovered posterior?}
                \item We generally don't have access to the true posterior
                      $P(\theta | D, M)$ -- only $P_\epsilon(\theta | D, M_\epsilon)$
                \item We need a \textbf{principled, instrument-aware metric}
            \end{itemize}
        \end{column}
        \hfill
        \begin{column}{0.42\textwidth}
            \centering
            \includegraphics[width=\textwidth]{figs/accuracy_comparison_ares_emulators_val_data.png}
            \\[0.3em]
            {\small Emulator error in signal space:\\
            with (orange) and without (purple) preprocessing}
        \end{column}
    \end{columns}
\end{frame}

\sectionframe{A KL divergence bound on posterior accuracy}

% ── 4. Setting up ─────────────────────────────────────────────────────────────
\begin{frame}{Emulator error propagates into the posterior}
    \begin{itemize}
        \item Emulator error $\epsilon(\theta) = M_\epsilon(\theta) - M(\theta)$
              perturbs the log-likelihood:
              \[
                  \log L \;\longrightarrow\; \log L + \delta \log L
              \]
        \item The posterior shifts from $P(\theta|D,M)$ to $P_\epsilon(\theta|D,M_\epsilon)$:
              \[
                  P_\epsilon = \frac{L\,\pi\,e^{\delta \log L}}
                  {\int L\,\pi\,e^{\delta \log L}\,d\theta}
              \]
        \item A natural measure of the difference is the
              \textbf{Kullback--Leibler divergence}:
              \[
                  \mathcal{D}_\mathrm{KL}(P \| P_\epsilon)
                  = \int P \log\!\left(\frac{P}{P_\epsilon}\right) d\theta
              \]
        \item This quantifies the \textbf{incorrect information inferred} (in nats)
              when using the emulator --- we want $\mathcal{D}_\mathrm{KL} < 1$
        \item Problem: we don't have access to $P$ to evaluate this directly \ldots
    \end{itemize}
\end{frame}

% ── 5. Main result ────────────────────────────────────────────────────────────
\begin{frame}{A theoretically motivated upper bound}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            Under three assumptions:
            \begin{itemize}
                \item \textbf{Gaussian likelihood} and broad uniform prior
                \item \textbf{Linear model} around the posterior peak
                \item \textbf{White (uncorrelated) noise} in the data
            \end{itemize}
            \vspace{0.2em}
            \ldots analytically:
            \[
                \boxed{
                    \mathcal{D}_\mathrm{KL}(P \| P_\epsilon)
                    \leq \frac{N_d}{2}
                    \left(\frac{\mathrm{RMSE}}{\sigma}\right)^{\!2}
                }
            \]
            \vspace{-0.3em}
            For $\mathcal{D}_\mathrm{KL} < 1$:
            $\;\frac{\mathrm{RMSE}}{\sigma} \leq \sqrt{\frac{2}{N_d}}$
            \;($\lesssim 14\%$ for $N_d \sim 100$)
        \end{column}
        \hfill
        \begin{column}{0.42\textwidth}
            \centering
            \includegraphics[width=\textwidth]{figs/kl_divergence.png}
            \\[0.3em]
            {\small KL divergence bound as a function of noise $\sigma$
            and emulator RMSE.
            Crosses show our measured $\mathcal{D}_\mathrm{KL}$ values.}
        \end{column}
    \end{columns}
\end{frame}

% ── 6. What this gives us ─────────────────────────────────────────────────────
\begin{frame}{Why this bound is useful}
    \begin{itemize}
        \item \textbf{Instrument-aware}: the required emulator accuracy scales with the
              noise -- better instruments demand more accurate emulators
              \begin{itemize}
                  \item For $N_d = 100$: $\mathrm{RMSE}/\sigma \lesssim 14\%$ for
                        $\mathcal{D}_\mathrm{KL} < 1$
                  \item SKA power spectrum: $N_d \sim 10^3$--$10^4$, so requirements
                        tighten as $\sqrt{2/N_d}$
              \end{itemize}
        \item \textbf{Predictive}: evaluable without running the full simulation in
              inference -- only RMSE on a test set and $\sigma$ are needed
        \item \textbf{Can be inverted}: given a target posterior accuracy and noise
              level, compute the required RMSE \emph{before} running inference
        \item \textbf{Loss function / hyperparameter tuning}: optimise emulator
              training to minimise the bound directly, instrument by instrument
        \item \textbf{Broadly applicable}: any pipeline with a Gaussian likelihood
              and approximately linear model -- SKA data analysis, SED fitting, CMB, \ldots
    \end{itemize}
\end{frame}

% ── B1. Derivation sketch ─────────────────────────────────────────────────────
\begin{frame}{Derivation: linear model and Gaussian likelihood}
    \begin{itemize}
        \item Assume linear model $\mathcal{M}(\theta) = M\theta + m$ and
              linear emulator error $\epsilon(\theta) = E\theta + \epsilon_0$
        \item For $E \ll M$ (slowly varying emulator error), the emulated
              posterior covariance $C_\epsilon \approx C$ and the mean shifts by:
              \[
                  \mu_\epsilon \approx \mu - C M^T \Sigma^{-1} \epsilon_0
              \]
        \item The KL divergence between two Gaussians then simplifies to:
              \[
                  \mathcal{D}_\mathrm{KL} = \tfrac{1}{2}\,
                  \epsilon_0^T \Sigma^{-1} M \!\left(M^T \Sigma^{-1} M\right)^{-1}\!
                  M^T \Sigma^{-1} \epsilon_0
              \]
        \item For white noise ($\Sigma = \sigma^2 \mathbf{1}$) this is
              $\frac{1}{2\sigma^2}\,\epsilon_0^T M M^+ \epsilon_0$
              where $M M^+$ is a projection matrix with eigenvalues in $\{0,1\}$
        \item Bounding via $\max(\lambda_n) = 1$ and $\|\epsilon_0\|^2 = N_d\,\mathrm{RMSE}^2$
              gives the result on slide 5
    \end{itemize}
\end{frame}

% ── B2. Limitations ───────────────────────────────────────────────────────────
\begin{frame}{Limitations of the approximation}
    \begin{itemize}
        \item Assumes \textbf{linearity} of the model around the posterior peak --
              may break down in high dimensions or highly curved posteriors
        \item Assumes \textbf{Gaussian likelihood} -- may not hold for all
              instruments or data preprocessing choices
        \item Assumes \textbf{uncorrelated (white) noise} -- extension to
              correlated noise is straightforward if $\Sigma$ is known
        \item The bound is \textbf{conservative}: measured
              $\mathcal{D}_\mathrm{KL}$ values are typically well below the bound
        \item Applies most directly to posteriors that are \textbf{unimodal and
              approximately Gaussian} -- multimodal cases warrant extra care
    \end{itemize}
\end{frame}
